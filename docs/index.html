Three ways to program

Load .il files into Virtuoso, execute any SKILL expression, or use Python APIs for layout, schematic, and Spectre simulation.

AI-native design

CLI-first, ships with agent skill files, persistent SSH for high-frequency interactions, full command logging.

30+ runnable examples: layout editing, schematic creation, DC/AC/transient/PSS/Pnoise simulation.

Architecture

Quick Start

pip install -e .
virtuoso-bridge init        # generates .env
# edit .env → set VB_REMOTE_HOST=my-server
virtuoso-bridge start
from virtuoso_bridge import VirtuosoClient

client = VirtuosoClient.from_env()
result = client.execute_skill("1+2")
print(result)  # {'ok': True, 'result': {'output': '3', ...}}

Layout editing

with client.layout.edit("myLib", "myCell") as layout:
    layout.add_rect("M1", "drawing", (0, 0, 1, 0.5))
    layout.add_path("M2", "drawing", [(0,0),(1,0)], width=0.1)
    layout.add_label("M1", "pin", (0.5, 0.25), "VDD")
    layout.add_polygon("M3", "drawing", [(0,0),(1,0),(1,1)])
    layout.add_instance("tsmcN28", "nch_ulvt_mac", (0,0), "M0")
    layout.add_via("M1_M2", (0.5, 0.25))
    layout.add_mosaic("tsmcN28", "nch_ulvt_mac", (0,0), rows=2, cols=4,
                      row_pitch=0.5, col_pitch=1.0)
    shapes = layout.get_shapes()

Schematic editing

with client.schematic.edit("myLib", "myCell") as sch:
    sch.add_instance("analogLib", "vdc", (0,0), "V0", params={"vdc": "0.9"})
    sch.add_wire([(0, 0), (0, 0.5)])
    sch.add_pin("VDD", "inputOutput", (0, 1.0))
    sch.add_net_label_to_instance_term("V0", "PLUS", "VDD")
    sch.add_wire_between_instance_terms("V0", "MINUS", "GND0", "gnd!")
    sch.add_pin_to_instance_term("V0", "PLUS", "VIN", "input")

Spectre simulation

from virtuoso_bridge.spectre.runner import SpectreSimulator

sim = SpectreSimulator.from_env(work_dir="./output")
result = sim.run_simulation("tb_inv.scs", {})

print(result.status)          # ExecutionStatus.SUCCESS
print(result.data.keys())     # ['time', 'VOUT', 'VIN', ...]
print(result.data["VOUT"][:5]) # first 5 voltage samples

# Check license before running
info = sim.check_license()
print(info["spectre_path"])    # /path/to/spectre
print(info["licenses"])        # license feature usage

ADE Maestro control

# Read design variables
client.execute_skill('asiGetDesignVarList(asiGetCurrentSession())')

# Set design variable
client.execute_skill('maeSetVar("VDD" "0.85")')

# Read enabled analyses for a test
client.execute_skill('maeGetEnabledAnalysis("tb_cmp_SA")')

# Read full PSS / Pnoise configuration
client.execute_skill('maeGetAnalysis("tb_cmp_SA" "pss")')
client.execute_skill('maeGetAnalysis("tb_cmp_SA" "pnoise")')

# Modify analysis options
client.execute_skill('maeSetAnalysis("tb_cmp_SA" "pnoise" ?options list(list("stop" "1G")))')

# Trigger simulation (async)
client.execute_skill('maeRunSimulation()')
client.execute_skill('maeWaitUntilDone(\'All)')

# Read results via OCEAN (built into CIW)
client.execute_skill('openResults("...")')
client.execute_skill('ocnPrint(v("/OUT") ?output "/tmp/out.txt")')
client.download_file('/tmp/out.txt', 'output/out.txt')

SSHClient — independent SSH layer

from virtuoso_bridge import SSHClient, VirtuosoClient

# Remote mode — create tunnel then connect
tunnel = SSHClient.from_env()
tunnel.warm()
bridge = VirtuosoClient.from_tunnel(tunnel)

# Local mode — no tunnel needed
bridge = VirtuosoClient.local(port=65432)
